Prefixo p/ arquivos:
Iniciar Upload | Limpar fila de Upload

Código:

var uploader = new multiUpload('uploader', 'uploader_files', {
	swf:            '../../swf/multiUpload.swf',
	script:         'upload.php',
	expressInstall: '../../swf/expressInstall.swf',
	multi:          true
});

function sendIt()
{
	var prefix = document.getElementById("txtPrefix").value;

	if (prefix.length)
	{
		uploader.setData({
			Prefix: prefix
		});
		uploader.startUpload();
	}
	else
		alert("Digite o prefixo.");
}

Ação "Enviar":

<a href="javascript:sendIt();">Iniciar Upload</a>
